Важно: всегда возвращать функцию очистки, иначе таймер продолжит работать после того как компонент удалён со страницы (утечка памяти).

Хук useCountdown

frontend/src/hooks/useCountdown.ts
import { useState, useEffect } from 'react'

interface TimeLeft {
  days:    number
  hours:   number
  minutes: number
  seconds: number
  expired: boolean
}

export function useCountdown(targetDate: Date): TimeLeft {
  const calc = (): TimeLeft => {
    const diff = targetDate.getTime() - Date.now()
    // Date.now() — миллисекунды с 1 января 1970 (unix timestamp)
    // targetDate.getTime() — то же самое для целевой даты
    // diff = сколько миллисекунд осталось

    if (diff <= 0) return { days: 0, hours: 0, minutes: 0, seconds: 0, expired: true }

    return {
      days:    Math.floor(diff / (1000 * 60 * 60 * 24)),
      hours:   Math.floor(diff / (1000 * 60 * 60) % 24),
      minutes: Math.floor(diff / (1000 * 60) % 60),
      seconds: Math.floor(diff / 1000 % 60),
      expired: false
    }
    // 1000 мс = 1 сек, × 60 = 1 мин, × 60 = 1 час, × 24 = 1 день
    // % 24 — остаток от деления: часы в рамках одних суток
  }

  const [timeLeft, setTimeLeft] = useState<TimeLeft>(calc)

  useEffect(() => {
    // setInterval вызывает функцию каждые 1000 мс (1 секунда)
    const id = setInterval(() => {
      setTimeLeft(calc())
    }, 1000)

    // Функция очистки — вызывается при размонтировании компонента
    // Без clearInterval таймер продолжит тикать и обновлять unmounted компонент
    return () => clearInterval(id)
  }, [targetDate])             // зависимость: если дата изменится — перезапустить

  return timeLeft
}

import { useCountdown } from '../hooks/useCountdown'

const SALE_END = new Date('2026-06-01T00:00:00')  // дата окончания акции

        </div>

💡 Почему каждую секунду перерисовывается только таймер?
React сравнивает предыдущий и новый Virtual DOM. Изменились только числа внутри .timer-num — именно эти 4 элемента и обновляются в реальном DOM. Всё остальное не трогается.

Хук useNotifications

Хук периодически проверяет сервер на новые уведомления с помощью polling.

frontend/src/hooks/useNotifications.ts
import { useState, useEffect, useCallback } from 'react'

interface Notification {
  id:      number
  message: string
  read:    boolean
}

export function useNotifications() {
  const [items,   setItems]   = useState<Notification[]>([])
  const [loading, setLoading] = useState(true)

  const fetch_notifs = useCallback(async () => {
    const token = localStorage.getItem('token')
    if (!token) return
    try {
      const res = await fetch('/api/notifications', {
        headers: { 'Authorization': `Bearer ${token}` }
      })
      if (res.ok) setItems(await res.json())
    } finally {
      setLoading(false)
    }
  }, [])

  useEffect(() => {
    fetch_notifs()                       // первая загрузка

    // Polling: проверять каждые 30 секунд
    const id = setInterval(fetch_notifs, 30_000)
    return () => clearInterval(id)     // очистить при unmount
  }, [fetch_notifs])

  // Пометить все как прочитанные
  const markAllRead = async () => {
    await fetch('/api/notifications/read-all', {
      method:  'PATCH',
      headers: { 'Authorization': `Bearer ${localStorage.getItem('token')}` }
    })
    setItems(prev => prev.map(n => ({ ...n, read: true })))  // optimistic
  }

  const unreadCount = items.filter(n => !n.read).length

  return { items, loading, unreadCount, markAllRead }
}

Компонент NotificationBell

frontend/src/components/NotificationBell.tsx
import { useState } from 'react'
import { useNotifications } from '../hooks/useNotifications'

export function NotificationBell() {
  const { items, unreadCount, markAllRead } = useNotifications()
  const [open, setOpen] = useState(false)

  const handleOpen = () => {
    setOpen(v => !v)
    if (unreadCount > 0) markAllRead()
  }

  return (
    <div className="notif-wrap">
      <button className="notif-btn" onClick={handleOpen}>
        🔔
        {unreadCount > 0 && (
          <span className="notif-badge">
            {unreadCount > 9 ? '9+' : unreadCount}
            {'// 9+ если больше 9, чтобы бейдж не переполнялся'}
          </span>
        )}
      </button>

      {open && (
        <div className="notif-dropdown">
          {items.length === 0
            ? <p className="notif-empty">Нет уведомлений</p>
            : items.map(n => (
                <div key={n.id} className={`notif-item ${n.read ? 'read' : 'unread'}`}>
                  {n.message}
                </div>
              ))
          }
        </div>
      )}
    </div>
  )
}

import { Router }      from 'express'
import { prisma }      from '../index'
import { requireAuth } from '../middleware/auth'

const router = Router()

router.get('/', requireAuth, async (req, res) => {
  const items = await prisma.notification.findMany({
    where:   { userId: req.userId },
    orderBy: { createdAt: 'desc' },
    take:    20         // не больше 20 последних
  })
  res.json(items)
})

router.patch('/read-all', requireAuth, async (req, res) => {
  await prisma.notification.updateMany({
    where:  { userId: req.userId, read: false },
    data:   { read: true }
  })
  res.json({ ok: true })
})

export default router

Роут для настроек

backend/src/routes/settings.ts
import { Router }      from 'express'
import { prisma }      from '../index'
import { requireAuth } from '../middleware/auth'

const router = Router()

// GET /api/settings — получить настройки текущего пользователя
router.get('/', requireAuth, async (req, res) => {
  const settings = await prisma.userSettings.findUnique({
    where: { userId: req.userId }
  })
  // Если записи нет — вернуть дефолтные настройки
  res.json(settings ?? { theme: 'dark' })
})

// PATCH /api/settings — обновить настройки
router.patch('/', requireAuth, async (req, res) => {
  const { theme } = req.body

  if (![ 'dark', 'light' ].includes(theme)) {
    return res.status(400).json({ error: 'theme должен быть dark или light' })
  }

  // upsert: создать если нет, обновить если есть
  const result = await prisma.userSettings.upsert({
    where:  { userId: req.userId! },
    update: { theme },
    create: { userId: req.userId!, theme }
  })

  res.json(result)
})

export default router

Хук useTheme — синхронизация с сервером

frontend/src/hooks/useTheme.ts
import { useState, useEffect } from 'react'

type Theme = 'dark' | 'light'

export function useTheme() {
  // Начинаем с localStorage — быстро, без задержки сервера
  const [theme, setTheme] = useState<Theme>(
    (localStorage.getItem('theme') as Theme) ?? 'dark'
  )

  // При монтировании загрузить тему с сервера (если пользователь вошёл)
  useEffect(() => {
    const token = localStorage.getItem('token')
    if (!token) return

    fetch('/api/settings', {
      headers: { 'Authorization': `Bearer ${token}` }
    })
      .then(r => r.json())
      .then(data => {
        applyTheme(data.theme)   // перетереть localStorage-значение серверным
      })
  }, [])

  const applyTheme = (t: Theme) => {
    document.documentElement.setAttribute('data-theme', t)
    localStorage.setItem('theme', t)
    setTheme(t)
  }

  const toggle = async () => {
    const next: Theme = theme === 'dark' ? 'light' : 'dark'
    applyTheme(next)           // мгновенно применить

    const token = localStorage.getItem('token')
    if (token) {
      // Сохранить на сервере в фоне
      await fetch('/api/settings', {
        method:  'PATCH',
        headers: {
          'Content-Type':  'application/json',
          'Authorization': `Bearer ${token}`
        },
        body: JSON.stringify({ theme: next })
      })
    }
  }

  return { theme, toggle }
}
💡 Стратегия: localStorage + сервер
1. При смене темы — сразу применяем (localStorage, никакой задержки)
2. В фоне — отправляем на сервер
3. При следующем входе — загружаем с сервера, перезаписываем localStorage
Пользователь не ждёт, данные синхронизированы между устройствами.
frontend/src/App.tsx — подключить useTheme
import { useTheme } from './hooks/useTheme'

function App() {
  const { theme, toggle } = useTheme()

  return (
    <div data-theme={theme}>   // CSS переменные реагируют на data-theme
      {'...'}
      <button onClick={toggle}>
        {theme === 'dark' ? '☀️ Светлая' : '🌙 Тёмная'}
      </button>
    </div>
  )
}
