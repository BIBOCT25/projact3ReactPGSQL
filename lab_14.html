Добавить поле isAdmin в модель User

Шаг 2 — Middleware

Добавить requireAdmin в auth.ts

Зачем два middleware: requireAuth проверяет только наличие токена (пользователь авторизован). requireAdmin дополнительно проверяет поле isAdmin из БД — нельзя стать администратором просто изменив payload токена.
📄 server/src/middleware/auth.ts — добавить после requireAuth
export const requireAdmin: RequestHandler = async (req, res, next) => {
  // requireAuth уже проверил токен и положил userId в req
  const user = await prisma.user.findUnique({
    where: { id: (req as any).userId }
  })
  if (!user?.isAdmin) {
    res.status(403).json({ error: 'Нет прав администратора' })
    return
  }
  next()
}
⚠️ Порядок middleware важен: на admin-маршрутах применяем requireAuth первым, затем requireAdmin. Если поменять местами — requireAdmin не найдёт userId в req.

Пример — эндпоинт статистики

📄 server/src/routes/admin.ts — GET /stats
router.get('/stats', async (req, res) => {
  const [totalProducts, totalOrders, totalUsers, revenueAgg] =
    await Promise.all([
      prisma.product.count(),
      prisma.order.count(),
      prisma.user.count(),
      prisma.order.aggregate({ _sum: { totalAmount: true } }),
    ])

  // Распределение по складу
  const out = await prisma.product.count({ where: { stock: 0 } })
  const low = await prisma.product.count({
    where: { stock: { gt: 0, lte: 5 } }
  })
  const ok  = await prisma.product.count({
    where: { stock: { gt: 5 } }
  })

  res.json({
    totalProducts, totalOrders, totalUsers,
    revenue: revenueAgg._sum.totalAmount ?? 0,
    stockDistrib: { out, low, ok },
    // ... topProducts, categoryStats, recentOrders
  })
})
💡 Совет: используйте Promise.all() для параллельных независимых запросов к БД — это значительно быстрее, чем последовательные await.

Пример — фильтр по остатку

📄 server/src/routes/admin.ts — GET /products
router.get('/products', async (req, res) => {
  const { page = 1, limit = 20, search, stock } = req.query

  // Формируем фильтр по остатку
  let stockFilter: any = {}
  if (stock === 'out') stockFilter = { stock: 0 }
  if (stock === 'low') stockFilter = { stock: { gt: 0, lte: 5 } }
  if (stock === 'ok')  stockFilter = { stock: { gt: 5 } }

  const where = {
    ...stockFilter,
    ...(search ? { name: { contains: search as string, mode: 'insensitive' } } : {})
  }

  const [products, total] = await Promise.all([
    prisma.product.findMany({
      where,
      include: { images: true, category: { select: { name: true } } },
      skip: (+page - 1) * +limit,
      take: +limit,
      orderBy: { id: 'desc' }
    }),
    prisma.product.count({ where })
  ])

  res.json({ products, total, page: +page, totalPages: Math.ceil(total / +limit) })
})

Эндпоинты категорий

📄 server/src/routes/admin.ts — CRUD категорий
// ─── КАТЕГОРИИ ──────────────────────────────────────────────────────────────

// GET /api/admin/categories
router.get('/categories', requireAuth, requireAdmin, async (req, res) => {
  const categories = await prisma.category.findMany({
    orderBy: { id: 'asc' },
    include: { _count: { select: { products: true } } }
  })
  res.json(categories)
})

// POST /api/admin/categories
router.post('/categories', requireAuth, requireAdmin, async (req, res) => {
  try {
    const { name, slug, parentId } = req.body
    if (!name || !slug) return res.status(400).json({ error: 'name и slug обязательны' })
    const category = await prisma.category.create({
      data: { name: String(name), slug: String(slug),
              parentId: parentId ? parseInt(parentId) : null }
    })
    res.json(category)
  } catch (e: any) { res.status(500).json({ error: e.message }) }
})

// PATCH /api/admin/categories/:id
router.patch('/categories/:id', requireAuth, requireAdmin, async (req, res) => {
  try {
    const id = parseInt(req.params.id)
    const { name, slug, parentId } = req.body
    const data: any = {}
    if (name !== undefined) data.name = String(name)
    if (slug !== undefined) data.slug = String(slug)
    if (parentId !== undefined) data.parentId = parentId ? parseInt(parentId) : null
    const category = await prisma.category.update({ where: { id }, data })
    res.json(category)
  } catch (e: any) { res.status(500).json({ error: e.message }) }
})

// DELETE /api/admin/categories/:id
router.delete('/categories/:id', requireAuth, requireAdmin, async (req, res) => {
  try {
    const id = parseInt(req.params.id)
    await prisma.category.delete({ where: { id } })
    res.json({ ok: true })
  } catch (e: any) { res.status(500).json({ error: e.message }) }
})
⚠️ Slug уникален: поле slug имеет ограничение @unique в схеме Prisma — при создании двух категорий с одинаковым slug сервер вернёт 500 с сообщением об ошибке. Генерируйте уникальные slug-и.

Шаг 5 — AuthContext

Добавить isAdmin в тип User

Зачем: TypeScript должен знать о поле isAdmin, иначе user?.isAdmin даст ошибку компиляции. Поле опциональное — старые ответы API без него не сломают логику.
📄 client/src/context/AuthContext.tsx
interface User {
  id: number
  email: string
  name?: string | null
  avatarUrl?: string | null
  isAdmin?: boolean        ← добавить
  settings?: { theme: string }
}

📄 client/src/pages/AdminPage.tsx
// Вкладки
type Tab = 'dashboard' | 'products' | 'orders' | 'users'

export function AdminPage() {
  const { user } = useAuth()
  const navigate = useNavigate()
  const [tab, setTab] = useState<Tab>('dashboard')

  // Редирект если не admin
  useEffect(() => {
    if (user !== null && !user?.isAdmin) navigate('/')
  }, [user, navigate])

  return (
    <div>
      {/* Вкладки */}
      {tab === 'dashboard' && <Dashboard />}
      {tab === 'products'  && <Products />}
      {tab === 'orders'    && <Orders />}
      {tab === 'users'     && <Users />}
    </div>
  )
}
