Часть 1 — Фича #25

🔍 FAQ — полнотекстовый поиск PostgreSQL

tsvector — это специальный тип данных PostgreSQL для полнотекстового поиска. Текст разбивается на «лексемы» (нормализованные слова), строится GIN-индекс. Поиск работает в сотни раз быстрее, чем LIKE '%слово%', и понимает словоформы.
1

Создать GIN-индекс в PostgreSQL

Выполнить в psql или Prisma Studio (SQL-вкладка):

-- Добавить колонку для поискового вектора
ALTER TABLE "FaqItem" ADD COLUMN
  search_vector tsvector;

-- Заполнить вектор для существующих строк
UPDATE "FaqItem"
  SET search_vector = to_tsvector(
    'russian',
    question || ' ' || answer
  );

-- Создать быстрый GIN-индекс
CREATE INDEX faq_search_idx
  ON "FaqItem" USING GIN(search_vector);

-- Триггер: автообновлять вектор при INSERT/UPDATE
CREATE OR REPLACE FUNCTION faq_search_update() RETURNS trigger AS $$
BEGIN
  NEW.search_vector := to_tsvector('russian', NEW.question || ' ' || NEW.answer);
  RETURN NEW;
END
$$ LANGUAGE plpgsql;

CREATE TRIGGER faq_search_trigger
  BEFORE INSERT OR UPDATE ON "FaqItem"
  FOR EACH ROW EXECUTE FUNCTION faq_search_update();
Prisma не управляет raw-SQL триггерами — добавь эти команды в отдельный prisma/migrations/XXX/migration_custom.sql или выполни вручную после prisma migrate dev.

Backend — GET /api/faq/search

backend/src/routes/faq.ts
import { Router } from 'express'
import { prisma } from '../index'

const router = Router()

router.get('/search', async (req, res) => {
  const { q = '' } = req.query as { q: string }

  if (!q.trim()) {
    // Без запроса — вернуть все FAQ
    const all = await prisma.faqItem.findMany({
      select: { id: true, question: true, answer: true }
    })
    return res.json(all)
  }

  // Полнотекстовый поиск с ранжированием по релевантности
  const results = await prisma.$queryRaw`
    SELECT id, question, answer,
           ts_rank(search_vector, plainto_tsquery('russian', ${q})) AS rank
    FROM   "FaqItem"
    WHERE  search_vector @@ plainto_tsquery('russian', ${q})
    ORDER BY rank DESC
  `
  res.json(results)
})

export default router
Параметризованный запрос: $queryRaw с template literals автоматически передаёт q как параметр — защита от SQL-инъекций.

React — FaqPage с live-поиском

frontend/src/pages/FaqPage.tsx
import { useState, useEffect } from 'react'

interface FaqItem { id: number; question: string; answer: string }

export function FaqPage() {
  const [query, setQuery] = useState('')
  const [items, setItems] = useState<FaqItem[]>([])
  const [open,  setOpen]  = useState<number | null>(null)

  useEffect(() => {
    const timer = setTimeout(async () => {
      const url = `/api/faq/search?q=${encodeURIComponent(query)}`
      const data = await fetch(url).then(r => r.json())
      setItems(data)
    }, 300)
    return () => clearTimeout(timer)
  }, [query])

  return (
    <div>
      <h2>Часто задаваемые вопросы</h2>
      <input
        value={query}
        onChange={e => setQuery(e.target.value)}
        placeholder="Поиск по FAQ..."
      />
      {items.map(item => (
        <div key={item.id}>
          <div
            onClick={() => setOpen(open === item.id ? null : item.id)}
            style={{ cursor: 'pointer', fontWeight: 600 }}
          >
            {open === item.id ? '▼' : '▶'} {item.question}
          </div>
          {open === item.id && <p>{item.answer}</p>}
        </div>
      ))}
      {!items.length && query && <p>Ничего не найдено</p>}
    </div>
  )
}

React — ProductGallery.tsx

import { useState } from 'react'

interface ProductImage { id: number; url: string; order: number }

export function ProductGallery({ images }: { images: ProductImage[] }) {
  const [active, setActive] = useState(0)

  if (!images.length) return <div className="no-image">Нет фото</div>

  return (
    <div style={{ display: 'flex', gap: 12 }}>

      {/* Главное фото */}
      <img
        src={images[active].url}
        alt="Товар"
        style={{ width: 320, height: 320, objectFit: 'cover', borderRadius: 10 }}
      />

      {/* Миниатюры */}
      <div style={{ display: 'flex', flexDirection: 'column', gap: 8 }}>
        {images.map((img, i) => (
          <img
            key={img.id}
            src={img.url}
            alt={`Фото ${i + 1}`}
            onClick={() => setActive(i)}
            style={{
              width:      64,
              height:     64,
              objectFit:  'cover',
              borderRadius: 6,
              cursor:     'pointer',
              border:     i === active ? '2px solid #2dd4bf' : '2px solid transparent',
              opacity:    i === active ? 1 : 0.7,
              transition: 'all .2s'
            }}
          />
        ))}
      </div>
    </div>
  )
}

Backend — POST /api/subscribe

backend/src/routes/subscribe.ts
import { Router } from 'express'
import { prisma } from '../index'

const router = Router()

router.post('/', async (req, res) => {
  const { email } = req.body

  if (!email?.includes('@'))
    return res.status(400).json({ error: 'Некорректный email' })

  // Проверить — вдруг уже подписан
  const exists = await prisma.subscriber.findUnique({ where: { email } })
  if (exists)
    return res.status(409).json({ error: 'Вы уже подписаны' })

  // Генерируем промокод для новичка
  const coupon = 'WELCOME' + Math.random().toString(36).slice(2, 6).toUpperCase()

  await prisma.subscriber.create({ data: { email, coupon } })

  // Здесь можно отправить письмо через Resend / SendGrid
  res.json({ ok: true, coupon })
})

export default router

React — SubscribeForm.tsx

frontend/src/components/SubscribeForm.tsx
import { useState } from 'react'

export function SubscribeForm() {
  const [email,  setEmail]  = useState('')
  const [coupon, setCoupon] = useState('')
  const [error,  setError]  = useState('')

  const submit = async (e: React.FormEvent) => {
    e.preventDefault()
    setError('')
    const res  = await fetch('/api/subscribe', {
      method:  'POST',
      headers: { 'Content-Type': 'application/json' },
      body:    JSON.stringify({ email })
    })
    const data = await res.json()
    if (!res.ok) { setError(data.error); return }
    setCoupon(data.coupon)
  }

  if (coupon) return (
    <div>
      <p>🎉 Вы подписаны! Ваш промокод:</p>
      <strong>{coupon}</strong>
    </div>
  )

  return (
    <form onSubmit={submit}>
      <h3>Подпишитесь и получите промокод</h3>
      <input
        type="email"
        value={email}
        onChange={e => setEmail(e.target.value)}
        placeholder="Ваш email"
        required
      />
      {error && <p style={{ color: '#ef4444' }}>{error}</p>}
      <button type="submit">Подписаться</button>
    </form>
  )
}
Часть 4 — GdprConsent

🍪 Баннер согласия на куки (GDPR)

GDPR — требует явного согласия пользователя на обработку данных. Для гостей: сохраняем guestId (UUID в localStorage). Для авторизованных: сохраняем userId.

router.post('/accept', async (req, res) => {
  const { guestId } = req.body
  const userId = req.userId ?? null   // null для гостей

  await prisma.gdprConsent.create({
    data: { userId, guestId: guestId ?? null }
  })
  res.json({ ok: true })
})

React — CookieBanner.tsx

frontend/src/components/CookieBanner.tsx
import { useState, useEffect } from 'react'

function getGuestId() {
  let id = localStorage.getItem('guestId')
  if (!id) {
    id = crypto.randomUUID()
    localStorage.setItem('guestId', id)
  }
  return id
}

export function CookieBanner() {
  const [show, setShow] = useState(false)

  useEffect(() => {
    // Показать если ещё не принял
    if (!localStorage.getItem('gdprAccepted')) setShow(true)
  }, [])

  const accept = async () => {
    localStorage.setItem('gdprAccepted', '1')
    setShow(false)
    await fetch('/api/gdpr/accept', {
      method:  'POST',
      headers: { 'Content-Type': 'application/json' },
      body:    JSON.stringify({ guestId: getGuestId() })
    })
  }

  if (!show) return null

  return (
    <div style={{
      position:   'fixed', bottom: 20, left: 20, right: 20,
      background: '#1c2230', border: '1px solid #30363d',
      borderRadius: 10, padding: '16px 20px',
      display: 'flex', justifyContent: 'space-between',
      alignItems: 'center', gap: 16, zIndex: 1000
    }}>
      <p>🍪 Мы используем куки для улучшения сайта.</p>
      <button
        onClick={accept}
        style={{ background: '#2dd4bf', color: '#fff', border: 'none', padding: '8px 20px', borderRadius: 8, cursor: 'pointer', fontWeight: 600 }}
      >
        Принять
      </button>
    </div>
  )
}
