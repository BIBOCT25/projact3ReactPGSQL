Роут проверки промокода

backend/src/routes/promo.ts
import { Router } from 'express'

// CONFIG-pattern: все промокоды в одном объекте
const PROMO_CODES: Record<string, { discount: number; label: string }> = {
  'SAVE10': { discount: 10, label: 'Скидка 10%' },
  'SAVE20': { discount: 20, label: 'Скидка 20%' },
  'FIRST50': { discount: 50, label: 'Первая покупка: −50%' },
}

const router = Router()

// POST /api/promo/check
router.post('/check', (req, res) => {
  const code = (req.body.code ?? '').trim().toUpperCase()
  // trim() убирает пробелы, toUpperCase() делает нечувствительным к регистру

  const promo = PROMO_CODES[code]

  if (!promo) {
    return res.status(404).json({ error: 'Промокод не найден' })
  }

  res.json(promo)
  // Ответ: { discount: 20, label: 'Скидка 20%' }
})

export default router

// В index.ts добавить:
import promoRouter from './routes/promo'
app.use('/api/promo', promoRouter)

CSS-анимация shake

Анимация «тряски» — классический UX-паттерн для ошибки ввода. Браузер запускает её добавлением класса.

/* В index.css */
@keyframes shake {
  0%,100% { transform: translateX(0);    }
  20%     { transform: translateX(-6px); }
  40%     { transform: translateX(6px);  }
  60%     { transform: translateX(-4px); }
  80%     { transform: translateX(4px);  }
}

.shake {
  animation: shake 0.45s ease;

frontend/src/components/PromoInput.tsx
import { useState, useRef } from 'react'

interface Props {
  onApply: (discount: number, label: string) => void
}

export function PromoInput({ onApply }: Props) {
  const [code,    setCode]    = useState('')
  const [error,   setError]   = useState('')
  const [success, setSuccess] = useState('')
  const inputRef = useRef<HTMLInputElement>(null)
  // useRef — ссылка на DOM-элемент (без перерендера)

  const handleApply = async () => {
    setError('')
    setSuccess('')

    const res = await fetch('/api/promo/check', {
      method:  'POST',
      headers: { 'Content-Type': 'application/json' },
      body:    JSON.stringify({ code })
    })

    if (!res.ok) {
      setError('Промокод не найден')

      // Запустить анимацию: добавить класс → убрать → добавить снова если нужно
      const el = inputRef.current
      if (el) {
        el.classList.remove('shake')
        // void нужен чтобы браузер заметил что класс убрали, перед тем как добавить снова
        void el.offsetWidth  // reflow trick
        el.classList.add('shake')
      }
      return
    }

    const { discount, label } = await res.json()
    setSuccess(label)
    onApply(discount, label)  // передать скидку наверх в CartPage
  }

  return (
    <div className="promo-wrap">
      <div className="promo-row">
        <input
          ref={inputRef}                  // ref подключаем к инпуту
          placeholder="Промокод"
          value={code}
          onChange={e => setCode(e.target.value)}
        />
        <button onClick={handleApply}>Применить</button>
      </div>

      {error   && <p className="promo-error">❌ {error}</p>}
      {success && <p className="promo-ok">✅ {success}</p>}
    </div>
  )
}
💡 void el.offsetWidth — reflow trick
Браузер оптимизирует: убрать и тут же добавить класс — он «не замечает» изменения. Обращение к offsetWidth заставляет браузер «сбросить» стили прямо сейчас (reflow), и анимация запускается заново.

Модель Rating в Prisma

prisma/schema.prisma — добавить
model Rating {
  id        Int      @id @default(autoincrement())
  score     Int      // 1–5
  userId    Int
  productId Int
  user      User     @relation(fields: [userId],    references: [id])
  product   Product  @relation(fields: [productId], references: [id])

  @@unique([userId, productId])
  // @@unique — комбинация userId+productId уникальна
  // = один пользователь ставит одну оценку одному товару
}

// В модели Product добавить обратную связь:
model Product {
  // ... существующие поля ...
  ratings Rating[]
}

Роут сохранения и получения рейтинга

backend/src/routes/ratings.ts
import { Router }      from 'express'
import { prisma }      from '../index'
import { requireAuth } from '../middleware/auth'

const router = Router()

// POST /api/ratings — поставить оценку (требует авторизации)
router.post('/', requireAuth, async (req, res) => {
  const { productId, score } = req.body
  const userId = req.userId!

  // Валидация: score должен быть 1–5
  if (score < 1 || score > 5 || !Number.isInteger(score)) {
    return res.status(400).json({ error: 'score должен быть 1–5' })
  }

  // upsert = UPDATE если запись есть, INSERT если нет
  // Это атомарная операция — не нужны два отдельных запроса
  await prisma.rating.upsert({
    where:  { userId_productId: { userId, productId } }, // условие поиска
    update: { score },                                    // что обновить
    create: { userId, productId, score }                  // что создать, если не нашли
  })

  // Вернуть обновлённый средний рейтинг
  const agg = await prisma.rating.aggregate({
    where:  { productId },
    _avg:   { score: true },   // вычислить среднее поля score
    _count: { score: true }    // посчитать количество оценок
  })

  res.json({
    avg:   +(agg._avg.score ?? 0).toFixed(1),  // округлить до 1 знака
    count: agg._count.score
  })
})

// GET /api/ratings/:productId — получить средний рейтинг
router.get('/:productId', async (req, res) => {
  const productId = +req.params.productId

  const agg = await prisma.rating.aggregate({
    where: { productId },
    _avg:  { score: true },
    _count:{ score: true }
  })

  res.json({
    avg:   +(agg._avg.score ?? 0).toFixed(1),
    count: agg._count.score
  })
})

export default router

Компонент StarRating

Интерактивные звёзды: наводим — подсвечиваются, кликаем — сохраняем оценку.

frontend/src/components/StarRating.tsx
import { useState, useEffect } from 'react'

interface Props {
  productId: number
  readonly?: boolean   // если true — только отображение без клика
}

export function StarRating({ productId, readonly = false }: Props) {
  const [avg,    setAvg]    = useState(0)
  const [count,  setCount]  = useState(0)
  const [hover,  setHover]  = useState(0)  // над какой звездой мышь
  const [myRate, setMyRate] = useState(0)  // моя текущая оценка

  // Загрузить рейтинг при монтировании компонента
  useEffect(() => {
    fetch(`/api/ratings/${productId}`)
      .then(r => r.json())
      .then(d => { setAvg(d.avg); setCount(d.count) })
  }, [productId])

  const handleClick = async (score: number) => {
    if (readonly) return
    setMyRate(score)   // optimistic update

    const res = await fetch('/api/ratings', {
      method:  'POST',
      headers: {
        'Content-Type':  'application/json',
        'Authorization': `Bearer ${localStorage.getItem('token')}`
      },
      body: JSON.stringify({ productId, score })
    })
    if (res.ok) {
      const updated = await res.json()
      setAvg(updated.avg)
      setCount(updated.count)
    }
  }

  // Какую звезду подсветить: hover имеет приоритет над myRate
  const filled = hover || myRate || Math.round(avg)

  return (
    <div className="star-rating">
      <div className="stars">
        {[1,2,3,4,5].map(n => (
          <span
            key={n}
            className={n <= filled ? 'star on' : 'star off'}
            onMouseEnter={() => !readonly && setHover(n)}
            onMouseLeave={() => setHover(0)}
            onClick={() => handleClick(n)}
          >★</span>
        ))}
      </div>
      <span className="star-info">
        {avg > 0 ? `${avg} (${count})` : 'Нет оценок'}
      </span>
    </div>
  )
}
💡 Почему upsert, а не два запроса?
Если делать findFirst + create/update — между ними может успеть прийти второй запрос и создать дублирующую запись (race condition). upsert атомарен: БД сама решает создать или обновить, без риска дубликатов.
