Часть 1 — Category модель

🗂 Категории товаров

Проблема без категорий: каталог из 500 товаров — это просто список без навигации. Категории дают структуру: «Одежда → Футболки», «Обувь → Кроссовки». PostgreSQL + Prisma позволяют строить дерево категорий через parentId — одна таблица, бесконечная вложенность.

Демо — фильтрация по категориям:

Все 👗 Одежда 👟 Обувь 💼 Аксессуары 🎒 Сумки

Обновить GET /api/products — добавить фильтр по категории:

router.get('/', async (req, res) => {
  const { page = '1', limit = '12', categoryId, q } = req.query as Record<string, string>
  const skip = (Number(page) - 1) * Number(limit)

  const where = {
    ...(categoryId ? { categoryId: Number(categoryId) } : {}),
    ...(q ? {
      name: { contains: q, mode: 'insensitive' as const }
    } : {})
  }

  const [products, total] = await Promise.all([
    prisma.product.findMany({
      where,
      skip, take: Number(limit),
      include: { category: { select: { name: true, slug: true } } }
    }),
    prisma.product.count({ where })
  ])

  res.json({ products, total, hasMore: skip + products.length < total })
})

React — CategoryNav + хлебные крошки

frontend/src/components/CategoryNav.tsx
import { useState, useEffect } from 'react'
import { useSearchParams } from 'react-router-dom'

interface Category { id: number; name: string; slug: string; children: Category[] }

export function CategoryNav() {
  const [cats, setCats]           = useState<Category[]>([])
  const [searchParams, setParams] = useSearchParams()
  const active = searchParams.get('categoryId')

  useEffect(() => {
    fetch('/api/categories').then(r => r.json()).then(setCats)
  }, [])

  const select = (id: number | null) => {
    const next = new URLSearchParams(searchParams)
    if (id) next.set('categoryId', String(id)); else next.delete('categoryId')
    next.delete('page')   // сбросить пагинацию при смене категории
    setParams(next)
  }

  return (
    <div style={{ display: 'flex', gap: 8, flexWrap: 'wrap', marginBottom: 20 }}>
      <button
        onClick={() => select(null)}
        style={{ background: !active ? '#34d399' : 'transparent' }}
      >Все</button>
      {cats.map(cat => (
        <button
          key={cat.id}
          onClick={() => select(cat.id)}
          style={{ background: active === String(cat.id) ? '#34d399' : 'transparent' }}
        >
          {cat.name}
        </button>
      ))}
    </div>
  )
}
useSearchParams — встроенный хук React Router. Хранит фильтры в URL (/products?categoryId=2&page=1). Пользователь может поделиться ссылкой на отфильтрованный каталог.

import { Router } from 'express'
import { prisma } from '../index'
import { requireAuth } from '../middleware/auth'

const router = Router()
router.use(requireAuth)   // все маршруты требуют авторизации

// Список адресов пользователя
router.get('/', async (req, res) => {
  const list = await prisma.address.findMany({
    where:   { userId: req.userId },
    orderBy: { isDefault: 'desc' }   // default — первым
  })
  res.json(list)
})

// Создать адрес
router.post('/', async (req, res) => {
  const { label, city, street, building, apartment, zip, isDefault } = req.body

  // Если новый адрес — default, снять флаг с остальных
  if (isDefault) {
    await prisma.address.updateMany({
      where: { userId: req.userId },
      data:  { isDefault: false }
    })
  }

  const addr = await prisma.address.create({
    data: { userId: req.userId!, label, city, street, building, apartment, zip, isDefault: !!isDefault }
  })
  res.status(201).json(addr)
})

// Удалить адрес (только свой)
router.delete('/:id', async (req, res) => {
  const addr = await prisma.address.findUnique({ where: { id: Number(req.params.id) } })
  if (!addr || addr.userId !== req.userId)
    return res.status(404).json({ error: 'Не найден' })

  await prisma.address.delete({ where: { id: addr.id } })
  res.json({ ok: true })
})

export default router

Часть 3 — Атомарный остаток

📦 Декремент остатка товара при заказе

Гонка данных (race condition): если два пользователя одновременно покупают последний товар, оба могут пройти проверку if (stock > 0) и оба создадут заказ. Решение: транзакция + атомарный декремент прямо в SQL.

Демо — индикаторы остатка:

✅ В наличии (23 шт.) ⚠️ Заканчивается (2 шт.) ❌ Нет в наличии
9

Транзакция: заказ + декремент остатка

Весь процесс оформления заказа должен быть одной транзакцией — либо всё, либо ничего.

backend/src/routes/orders.ts
router.post('/', requireAuth, async (req, res) => {
  try {
    const result = await prisma.$transaction(async (tx) => {

      // 1. Получить корзину
      const cartItems = await tx.cartItem.findMany({
        where:   { userId: req.userId },
        include: { product: true }
      })
      if (!cartItems.length) throw new Error('Корзина пуста')

      // 2. Проверить остатки + атомарный декремент
      for (const item of cartItems) {
        if (item.product.stock < item.qty) {
          throw new Error(`"${item.product.name}": недостаточно товара (осталось ${item.product.stock})`)
        }
        // decrement — атомарная операция, защита от гонки
        await tx.product.update({
          where: { id: item.productId },
          data:  { stock: { decrement: item.qty } }
        })
      }

      // 3. Посчитать итог
      const delivery  = req.body.deliveryRateId
        ? await tx.deliveryRate.findUnique({ where: { id: req.body.deliveryRateId } })
        : null
      const subtotal  = cartItems.reduce((s, i) => s + i.product.price * i.qty, 0)
      const total     = subtotal + (delivery?.price ?? 0)

      // 4. Создать заказ с позициями
      const order = await tx.order.create({
        data: {
          userId:          req.userId!,
          total,
          deliveryRateId:  req.body.deliveryRateId ?? null,
          addressId:       req.body.addressId      ?? null,
          addressSnapshot: req.body.addressSnapshot ?? null,
          items: {
            create: cartItems.map(i => ({
              productId: i.productId,
              qty:       i.qty,
              price:     i.product.price   // снимок цены
            }))
          }
        }
      })

      // 5. Очистить корзину
      await tx.cartItem.deleteMany({ where: { userId: req.userId } })

      return order
    })

    res.status(201).json(result)

  } catch (err: any) {
    // Транзакция автоматически откатилась — остатки не изменились
    res.status(400).json({ error: err.message })
  }
})
Prisma $transaction оборачивает всё в одну PostgreSQL-транзакцию. Если на шаге 2 остатка не хватает — вся транзакция откатывается, декремент отменяется, заказ не создаётся.

export function StockBadge({ stock }: { stock: number }) {
  if (stock === 0)  return <span className="stock-out">❌ Нет в наличии</span>
  if (stock <= 3)  return <span className="stock-low">⚠️ Осталось {stock} шт.</span>
  return             <span className="stock-ok">✅ В наличии</span>
}
