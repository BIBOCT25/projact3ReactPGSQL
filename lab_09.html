🛒 Оформление заказа (Order)

Что происходит при заказе: создаём запись Order со снимком товаров и цен (на случай изменения каталога), очищаем корзину, создаём уведомление пользователю.

Prisma — модели Order и OrderItem

prisma/schema.prisma — добавить
model Order {
  id         Int         @id @default(autoincrement())
  userId     Int
  total      Float
  address    String
  status     String      @default("pending")   // pending | paid | shipped | done
  createdAt  DateTime    @default(now())
  user       User        @relation(fields: [userId], references: [id])
  items      OrderItem[]
}

model OrderItem {
  id        Int     @id @default(autoincrement())
  orderId   Int
  productId Int
  name      String   // снимок названия на момент заказа
  price     Float    // снимок цены на момент заказа
  qty       Int
  order     Order   @relation(fields: [orderId], references: [id])
}

// В модели User добавить:
// orders  Order[]
Почему храним name/price в OrderItem? Если товар будет переименован или цена изменится — история заказа останется корректной.

Роут POST /api/orders

backend/src/routes/orders.ts
import { Router } from 'express'
import { prisma } from '../index'
import { requireAuth } from '../middleware/auth'

const router = Router()

// POST /api/orders — создать заказ из корзины
router.post('/', requireAuth, async (req, res) => {
  const { address } = req.body
  if (!address?.trim())
    return res.status(400).json({ error: 'Укажите адрес доставки' })

  // Загрузить корзину
  const cartItems = await prisma.cartItem.findMany({
    where: { userId: req.userId },
    include: { product: true }
  })
  if (!cartItems.length)
    return res.status(400).json({ error: 'Корзина пуста' })

  const total = cartItems.reduce((sum, i) => sum + i.product.price * i.qty, 0)

  // Транзакция: создать заказ + очистить корзину
  const order = await prisma.$transaction(async (tx) => {
    const newOrder = await tx.order.create({
      data: {
        userId: req.userId!,
        total,
        address,
        items: {
          create: cartItems.map(i => ({
            productId: i.productId,
            name:  i.product.name,
            price: i.product.price,
            qty:   i.qty
          }))
        }
      },
      include: { items: true }
    })

    // Очистить корзину
    await tx.cartItem.deleteMany({ where: { userId: req.userId } })

    // Создать уведомление
    await tx.notification.create({
      data: {
        userId:  req.userId!,
        message: `Заказ #${newOrder.id} на ${total} ₽ принят`
      }
    })

    return newOrder
  })

  res.status(201).json(order)
})

// GET /api/orders — история заказов
router.get('/', requireAuth, async (req, res) => {
  const orders = await prisma.order.findMany({
    where: { userId: req.userId },
    include: { items: true },
    orderBy: { createdAt: 'desc' }
  })
  res.json(orders)
})

export default router
prisma.$transaction() — атомарная операция. Если любой шаг упадёт — все изменения откатятся. Заказ создастся только если корзина успешно очищена.

CheckoutPage.tsx — страница оформления

frontend/src/pages/CheckoutPage.tsx
import { useState } from 'react'
import { useNavigate } from 'react-router-dom'
import { useCart } from '../hooks/useCart'

export function CheckoutPage() {
  const { cart } = useCart()
  const navigate = useNavigate()
  const [address, setAddress] = useState('')
  const [loading, setLoading] = useState(false)
  const [error, setError]     = useState('')

  const submit = async (e: React.FormEvent) => {
    e.preventDefault()
    setLoading(true)
    setError('')

    const res = await fetch('/api/orders', {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'Authorization': `Bearer ${localStorage.getItem('token')}`
      },
      body: JSON.stringify({ address })
    })

    setLoading(false)
    if (!res.ok) {
      const data = await res.json()
      setError(data.error)
    } else {
      navigate('/orders')   // перейти в историю заказов
    }
  }

  return (
    <form onSubmit={submit}>
      <h2>Оформление заказа</h2>
      <p>Сумма: <strong>{cart?.total} ₽</strong></p>
      <input
        value={address}
        onChange={e => setAddress(e.target.value)}
        placeholder="Улица, дом, квартира"
        required
      />
      {error && <p style={{color:'red'}}>{error}</p>}
      <button type="submit" disabled={loading}>
        {loading ? 'Оформляем...' : 'Подтвердить заказ'}
      </button>
    </form>
  )
}

OrdersPage.tsx — история заказов

frontend/src/pages/OrdersPage.tsx
import { useState, useEffect } from 'react'

interface OrderItem { name: string; price: number; qty: number }
interface Order     { id: number; total: number; status: string; address: string; createdAt: string; items: OrderItem[] }

export function OrdersPage() {
  const [orders, setOrders] = useState<Order[]>([])
  const [open, setOpen]     = useState<number | null>(null)

  useEffect(() => {
    fetch('/api/orders', {
      headers: { 'Authorization': `Bearer ${localStorage.getItem('token')}` }
    }).then(r => r.json()).then(setOrders)
  }, [])

  const STATUS: Record<string, string> = {
    pending:  '⏳ Принят',
    paid:     '💳 Оплачен',
    shipped:  '🚚 В пути',
    done:     '✅ Доставлен',
  }

  if (!orders.length) return <p>Заказов ещё нет</p>

  return (
    <div>
      <h2>Мои заказы</h2>
      {orders.map(order => (
        <div key={order.id}>
          <div onClick={() => setOpen(open === order.id ? null : order.id)}>
            <span>Заказ #{order.id}</span>
            <span>{order.total} ₽</span>
            <span>{STATUS[order.status]}</span>
            <span>{new Date(order.createdAt).toLocaleDateString('ru')}</span>
          </div>
          {open === order.id && (
            <div>
              <p>Адрес: {order.address}</p>
              {order.items.map((item, i) => (
                <div key={i}>{item.name} × {item.qty} = {item.price * item.qty} ₽</div>
              ))}
            </div>
          )}
        </div>
      ))}
    </div>
  )
}

Backend — расширенный GET /api/products

backend/src/routes/products.ts — обновить
router.get('/', async (req, res) => {
  const {
    q        = '',
    minPrice = '0',
    maxPrice = '999999',
    sort     = 'id_asc',
    page     = '1',
    limit    = '12'
  } = req.query as Record<string, string>

  const skip = (Number(page) - 1) * Number(limit)

  const where = {
    price: { gte: Number(minPrice), lte: Number(maxPrice) },
    ...(q ? { name: { contains: q, mode: 'insensitive' as const } } : {})
  }

  const orderBy: Record<string, Record<string, string>> = {
    id_asc:     { id:    'asc'  },
    price_asc:  { price: 'asc'  },
    price_desc: { price: 'desc' },
    name_asc:   { name:  'asc'  },
  }

  const [products, total] = await prisma.$transaction([
    prisma.product.findMany({
      where,
      orderBy: orderBy[sort] ?? { id: 'asc' },
      skip,
      take: Number(limit)
    }),
    prisma.product.count({ where })
  ])

  res.json({
    products,
    total,
    page:    Number(page),
    pages:   Math.ceil(total / Number(limit)),
    hasMore: skip + products.length < total
  })
})

useProducts.ts — хук с фильтрами

frontend/src/hooks/useProducts.ts
import { useState, useEffect, useCallback } from 'react'

export interface Filters {
  q:        string
  minPrice: number
  maxPrice: number
  sort:     string
}

export function useProducts(filters: Filters) {
  const [products, setProducts] = useState<any[]>([])
  const [page, setPage]         = useState(1)
  const [hasMore, setHasMore]   = useState(true)
  const [loading, setLoading]   = useState(false)

  const load = useCallback(async (p: number, reset = false) => {
    setLoading(true)
    const params = new URLSearchParams({
      q:        filters.q,
      minPrice: String(filters.minPrice),
      maxPrice: String(filters.maxPrice),
      sort:     filters.sort,
      page:     String(p),
      limit:    '12'
    })
    const data = await fetch(`/api/products?${params}`).then(r => r.json())
    setProducts(prev => reset ? data.products : [...prev, ...data.products])
    setHasMore(data.hasMore)
    setLoading(false)
  }, [filters])

  // Сбрасывать страницу при смене фильтров
  useEffect(() => { setPage(1); load(1, true) }, [filters])

  const loadMore = () => {
    const next = page + 1
    setPage(next)
    load(next)
  }

  return { products, hasMore, loading, loadMore }
}

FilterPanel.tsx — панель фильтров

frontend/src/components/FilterPanel.tsx
import { Filters } from '../hooks/useProducts'

interface Props {
  filters: Filters
  onChange: (f: Partial<Filters>) => void
  onReset: () => void
}

export function FilterPanel({ filters, onChange, onReset }: Props) {
  return (
    <aside>
      {/* Поиск */}
      <input
        value={filters.q}
        onChange={e => onChange({ q: e.target.value })}
        placeholder="Поиск товаров..."
      />

      {/* Диапазон цен */}
      <label>Цена от</label>
      <input type="range" min="0" max="10000"
        value={filters.minPrice}
        onChange={e => onChange({ minPrice: Number(e.target.value) })}
      />
      <span>{filters.minPrice} ₽</span>

      <label>Цена до</label>
      <input type="range" min="0" max="10000"
        value={filters.maxPrice}
        onChange={e => onChange({ maxPrice: Number(e.target.value) })}
      />
      <span>{filters.maxPrice} ₽</span>

      {/* Сортировка */}
      <select
        value={filters.sort}
        onChange={e => onChange({ sort: e.target.value })}
      >
        <option value="id_asc">По умолчанию</option>
        <option value="price_asc">Дешевле</option>
        <option value="price_desc">Дороже</option>
        <option value="name_asc">По названию</option>
      </select>

      <button onClick={onReset}>Сбросить фильтры</button>
    </aside>
  )
}
Паттерн controlled filters: все фильтры живут в состоянии родителя (CatalogPage), FilterPanel только отображает и вызывает onChange. Это делает состояние предсказуемым.

Backend — GET /api/products/:id

backend/src/routes/products.ts — добавить
router.get('/:id', async (req, res) => {
  const product = await prisma.product.findUnique({
    where: { id: Number(req.params.id) },
    include: {
      ratings: { select: { score: true } }  // для среднего рейтинга
    }
  })
  if (!product) return res.status(404).json({ error: 'Товар не найден' })

  const avgRating = product.ratings.length
    ? product.ratings.reduce((s, r) => s + r.score, 0) / product.ratings.length
    : null

  res.json({ ...product, avgRating, ratingCount: product.ratings.length })
})

ProductPage.tsx — детальная страница

frontend/src/pages/ProductPage.tsx
import { useState, useEffect } from 'react'
import { useParams, Link } from 'react-router-dom'
import { StarRating } from '../components/StarRating'

export function ProductPage() {
  const { id } = useParams()
  const [product, setProduct] = useState<any>(null)
  const [added, setAdded]     = useState(false)

  useEffect(() => {
    fetch(`/api/products/${id}`)
      .then(r => r.json())
      .then(setProduct)
  }, [id])

  const addToCart = async () => {
    await fetch('/api/cart', {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'Authorization': `Bearer ${localStorage.getItem('token')}`
      },
      body: JSON.stringify({ productId: Number(id) })
    })
    setAdded(true)
    setTimeout(() => setAdded(false), 2000)
  }

  if (!product) return <p>Загрузка...</p>

  return (
    <div>
      <p><Link to="/">← Каталог</Link></p>

      <h1>{product.name}</h1>
      <p>Цена: <strong>{product.price} ₽</strong></p>

      {product.avgRating && (
        <p>Рейтинг: ★ {product.avgRating.toFixed(1)} ({product.ratingCount} оценок)</p>
      )}

      <StarRating productId={product.id} />

      <button onClick={addToCart} disabled={added}>
        {added ? '✓ Добавлено' : 'В корзину'}
      </button>
    </div>
  )
}
