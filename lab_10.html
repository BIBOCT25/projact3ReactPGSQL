const BADGE_CONFIG: Record<string, { label: string; color: string }> = {

      fontWeight:   600,
      position:     'absolute',

      left:         8,
    }}>
      {label}
    </span>
  )
}

Часть 2 — Фича #16

✉️ Валидация email в реальном времени

Debounce + БД: пользователь вводит email → через 400ms после последнего нажатия запрос уходит на сервер → проверяет users на дубликат. Без debounce — запрос на каждый символ, нагрузка на БД.

Backend — GET /api/users/check-email

backend/src/routes/auth.ts — добавить
router.get('/check-email', async (req, res) => {
  const { email } = req.query as { email: string }

  // Базовая валидация — не делаем запрос в БД для явно невалидных значений
  if (!email || !email.includes('@'))
    return res.status(400).json({ error: 'Некорректный email' })

  const exists = await prisma.user.findUnique({ where: { email } })
  res.json({ taken: !!exists })
})

React — поле email с проверкой

frontend/src/components/EmailField.tsx
import { useState, useEffect } from 'react'

interface Props {
  value:    string
  onChange: (v: string) => void
}

export function EmailField({ value, onChange }: Props) {
  const [taken,   setTaken]   = useState(false)
  const [checking, setChecking] = useState(false)

  useEffect(() => {
    if (!value.includes('@')) { setTaken(false); return }

    setChecking(true)
    const timer = setTimeout(async () => {
      const res  = await fetch(`/api/auth/check-email?email=${encodeURIComponent(value)}`)
      const data = await res.json()
      setTaken(data.taken)
      setChecking(false)
    }, 400)                             // debounce 400ms

    return () => clearTimeout(timer)  // отмена предыдущего таймера
  }, [value])

  return (
    <div>
      <input
        type="email"
        value={value}
        onChange={e => onChange(e.target.value)}
        style={{ borderColor: taken ? '#ef4444' : '' }}
      />
      {checking && <span>Проверяем...</span>}
      {!checking && taken &&
        <span style={{ color: '#ef4444' }}>Email уже занят</span>
      }
      {!checking && !taken && value.includes('@') &&
        <span style={{ color: '#10b981' }}>✓ Email свободен</span>
      }
    </div>
  )
}
Паттерн cleanup в useEffect: функция возврата return () => clearTimeout(timer) отменяет предыдущий запрос при каждом новом нажатии клавиши. Без этого — запросы «гонятся» и последний может вернуться не последним.

🔔 Колокольчик уведомлений

Polling vs WebSocket: для большинства магазинов достаточно polling (проверка каждые 30 сек). WebSocket нужен если уведомления должны приходить в реальном времени (чат, торги).

import { prisma } from '../index'
import { requireAuth } from '../middleware/auth'

const router = Router()

// GET /api/notifications/unread-count
router.get('/unread-count', requireAuth, async (req, res) => {
  const count = await prisma.notification.count({
    where: { userId: req.userId, readAt: null }
  })
  res.json({ count })
})

// GET /api/notifications — последние 20
router.get('/', requireAuth, async (req, res) => {
  const items = await prisma.notification.findMany({
    where:   { userId: req.userId },
    orderBy: { createdAt: 'desc' },
    take:    20
  })
  res.json(items)
})

// PATCH /api/notifications/:id/read — пометить прочитанным
router.patch('/:id/read', requireAuth, async (req, res) => {
  const notif = await prisma.notification.findUnique({
    where: { id: Number(req.params.id) }
  })
  // Защита: нельзя читать чужие уведомления
  if (!notif || notif.userId !== req.userId)
    return res.status(403).json({ error: 'Нет доступа' })

  const updated = await prisma.notification.update({
    where: { id: Number(req.params.id) },
    data:  { readAt: new Date() }
  })
  res.json(updated)
})

// PATCH /api/notifications/read-all
router.patch('/read-all', requireAuth, async (req, res) => {
  await prisma.notification.updateMany({
    where: { userId: req.userId, readAt: null },
    data:  { readAt: new Date() }
  })
  res.json({ ok: true })
})

export default router
Уязвимость IDOR без проверки userId: всегда проверяй что уведомление принадлежит текущему пользователю. Без проверки любой авторизованный пользователь может прочитать чужие уведомления.

React — хук useNotifications

frontend/src/hooks/useNotifications.ts
import { useState, useEffect, useCallback } from 'react'

interface Notif {
  id:        number
  text:      string
  readAt:    string | null
  createdAt: string
}

export function useNotifications() {
  const [items,  setItems]  = useState<Notif[]>([])
  const [unread, setUnread] = useState(0)

  const auth = { headers: { 'Authorization': `Bearer ${localStorage.getItem('token')}` } }

  const fetchCount = useCallback(async () => {
    const data = await fetch('/api/notifications/unread-count', auth).then(r => r.json())
    setUnread(data.count)
  }, [])

  const fetchAll = useCallback(async () => {
    const data = await fetch('/api/notifications', auth).then(r => r.json())
    setItems(data)
    setUnread(data.filter((n: Notif) => !n.readAt).length)
  }, [])

  const markRead = async (id: number) => {
    await fetch(`/api/notifications/${id}/read`, { method: 'PATCH', ...auth })
    setItems(prev => prev.map(n => n.id === id ? { ...n, readAt: new Date().toISOString() } : n))
    setUnread(u => Math.max(0, u - 1))
  }

  const markAllRead = async () => {
    await fetch('/api/notifications/read-all', { method: 'PATCH', ...auth })
    fetchAll()
  }

  // Polling — каждые 30 секунд
  useEffect(() => {
    fetchCount()
    const id = setInterval(fetchCount, 30_000)
    return () => clearInterval(id)
  }, [])

  return { items, unread, fetchAll, markRead, markAllRead }
}

React — компонент NotificationBell

frontend/src/components/NotificationBell.tsx
import { useState } from 'react'
import { useNotifications } from '../hooks/useNotifications'

export function NotificationBell() {
  const { items, unread, fetchAll, markRead, markAllRead } = useNotifications()
  const [open, setOpen] = useState(false)

  const toggle = () => {
    if (!open) fetchAll()   // загрузить список при открытии
    setOpen(!open)
  }

  return (
    <div style={{ position: 'relative' }}>

      {/* Кнопка колокольчика с бейджем */}
      <button onClick={toggle} style={{ fontSize: 22, background: 'none', border: 'none', cursor: 'pointer', position: 'relative' }}>
        🔔
        {unread > 0 && (
          <span style={{
            position: 'absolute', top: -6, right: -8,
            background: '#ef4444', color: '#fff',
            borderRadius: '50%', width: 18, height: 18,
            display: 'flex', alignItems: 'center', justifyContent: 'center',
            fontSize: 11, fontWeight: 'bold'
          }}>
            {unread > 9 ? '9+' : unread}
          </span>
        )}
      </button>

      {/* Выпадающий список */}
      {open && (
        <div style={{ position: 'absolute', right: 0, top: '100%', width: 300, background: '#1c2230', border: '1px solid #30363d', borderRadius: 8, zIndex: 100 }}>
          <div style={{ display: 'flex', justifyContent: 'space-between', padding: '10px 14px', borderBottom: '1px solid #30363d' }}>
            <strong>Уведомления</strong>
            {unread > 0 && <button onClick={markAllRead}>Прочитать все</button>}
          </div>
          {items.map(n => (
            <div
              key={n.id}
              onClick={() => !n.readAt && markRead(n.id)}
              style={{ padding: '10px 14px', background: n.readAt ? 'transparent' : 'rgba(88,166,255,.1)', cursor: n.readAt ? 'default' : 'pointer', borderBottom: '1px solid #30363d' }}
            >
              {!n.readAt && <span style={{ display: 'inline-block', width: 8, height: 8, borderRadius: '50%', background: '#58a6ff', marginRight: 8 }}/>}
              {n.text}
            </div>
          ))}
          {!items.length && <p style={{ padding: 14, color: '#8b949e' }}>Нет уведомлений</p>}
        </div>
      )}
    </div>
  )
}

Backend — toggle + список избранного

backend/src/routes/favorites.ts
import { Router } from 'express'
import { prisma } from '../index'
import { requireAuth } from '../middleware/auth'

const router = Router()

// GET /api/favorites — список избранных товаров
router.get('/', requireAuth, async (req, res) => {
  const favs = await prisma.favorite.findMany({
    where:   { userId: req.userId },
    include: { product: true }
  })
  res.json(favs.map(f => f.product))
})

// POST /api/favorites/toggle
router.post('/toggle', requireAuth, async (req, res) => {
  const { productId } = req.body

  const existing = await prisma.favorite.findUnique({
    where: { userId_productId: { userId: req.userId!, productId } }
  })

  if (existing) {
    await prisma.favorite.delete({ where: { id: existing.id } })
    res.json({ favorited: false })
  } else {
    await prisma.favorite.create({ data: { userId: req.userId!, productId } })
    res.json({ favorited: true })
  }
})

export default router

React — FavoriteButton с оптимистичным обновлением

frontend/src/components/FavoriteButton.tsx
import { useState } from 'react'

interface Props {
  productId:        number
  initialFavorited: boolean
}

export function FavoriteButton({ productId, initialFavorited }: Props) {
  const [favorited, setFavorited] = useState(initialFavorited)
  const [loading,   setLoading]   = useState(false)

  const toggle = async () => {
    if (loading) return
    const prev = favorited

    // Оптимистичное обновление — меняем немедленно
    setFavorited(!favorited)
    setLoading(true)

    try {
      const res = await fetch('/api/favorites/toggle', {
        method:  'POST',
        headers: {
          'Content-Type':  'application/json',
          'Authorization': `Bearer ${localStorage.getItem('token')}`
        },
        body: JSON.stringify({ productId })
      })
      if (!res.ok) setFavorited(prev)  // откат при ошибке
    } catch {
      setFavorited(prev)              // откат при сетевой ошибке
    } finally {
      setLoading(false)
    }
  }

  return (
    <button
      onClick={toggle}
      disabled={loading}
      style={{
        fontSize:   22,
        background: 'none',
        border:     'none',
        cursor:     'pointer',
        color:      favorited ? '#ef4444' : '#8b949e',
        transition: 'color .2s'
      }}
    >
      {favorited ? '♥' : '♡'}
    </button>
  )
}

import { useState } from 'react'

export function NotifyStock({ productId }: { productId: number }) {
  const [show,  setShow]  = useState(false)
  const [email, setEmail] = useState('')
  const [sent,  setSent]  = useState(false)

  const submit = async () => {
    if (!email.includes('@')) return
    await fetch('/api/notify-stock', {
      method:  'POST',
      headers: { 'Content-Type': 'application/json' },
      body:    JSON.stringify({ email, productId })
    })
    setSent(true)
  }

  if (sent) return <p style={{ color: '#3fb950' }}>✅ Уведомим, когда появится</p>

  return (
    <div>
      <button onClick={() => setShow(!show)}>
        🔔 Уведомить о наличии
      </button>
      {show && (
        <div style={{ display: 'flex', gap: 8, marginTop: 8 }}>
          <input
            type="email"
            value={email}
            onChange={e => setEmail(e.target.value)}
            placeholder="Ваш email"
          />
          <button onClick={submit}>Отправить</button>
        </div>
      )}
    </div>
  )
}
