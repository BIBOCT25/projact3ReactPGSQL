Интеграция

🔗 Как всё соединить в App.tsx

frontend/src/App.tsx — итоговый
import { BrowserRouter, Routes, Route, Link } from 'react-router-dom'
import { AuthProvider }       from './context/AuthContext'
import { useAuth }           from './context/AuthContext'
import { useTheme }          from './hooks/useTheme'
import { ProtectedRoute }    from './components/ProtectedRoute'
import { CountdownBanner }  from './components/CountdownBanner'
import { NotificationBell } from './components/NotificationBell'
import { CatalogPage }      from './pages/CatalogPage'
import { CartPage }         from './pages/CartPage'
import { LoginPage }        from './pages/LoginPage'
import './index.css'

// NavBar вынесен в отдельную функцию — использует хуки
function NavBar() {
  const { token, logout } = useAuth()
  const { theme, toggle } = useTheme()

  return (
    <nav className="app-nav">
      <Link to="/">🛍 Магазин</Link>

      <div className="nav-right">
        <button onClick={toggle}>
          {theme === 'dark' ? '☀️' : '🌙'}
        </button>

        {token && <NotificationBell />}

        {token
          ? <>
              <Link to="/cart">🛒</Link>
              <button onClick={logout}>Выйти</button>
            </>
          : <Link to="/login">Войти</Link>
        }
      </div>
    </nav>
  )
}

function App() {
  return (
    <AuthProvider>
      <BrowserRouter>
        <CountdownBanner />  // баннер акции поверх всего
        <NavBar />

        <main className="app-main">
          <Routes>
            <Route path="/"      element={<CatalogPage />} />
            <Route path="/login" element={<LoginPage />}   />
            <Route path="/cart"
              element={
                <ProtectedRoute><CartPage /></ProtectedRoute>
              }
            />
          </Routes>
        </main>
      </BrowserRouter>
    </AuthProvider>
  )
}

export default App
Порядок провайдеров важен. AuthProvider снаружи — все компоненты внутри (включая NavBar) могут вызвать useAuth(). BrowserRouter снаружи — все компоненты могут вызвать useNavigate() и Link.
База данных

📐 Итоговая схема Prisma

// Полный schema.prisma — все модели

model Product {
  id        Int        @id @default(autoincrement())
  name      String
  price     Int
  imageUrl  String?
  cartItems CartItem[]
  ratings   Rating[]
}

model User {
  id           Int            @id @default(autoincrement())
  email        String         @unique
  passwordHash String
  createdAt    DateTime       @default(now())
  cartItems    CartItem[]
  ratings      Rating[]
  notifications Notification[]
  settings     UserSettings?
}

model CartItem {
  id        Int     @id @default(autoincrement())
  qty       Int     @default(1)
  productId Int
  userId    Int
  product   Product @relation(fields: [productId], references: [id])
  user      User    @relation(fields: [userId],    references: [id])
  @@unique([userId, productId])
}

model Rating {
  id        Int     @id @default(autoincrement())
  score     Int
  userId    Int
  productId Int
  user      User    @relation(fields: [userId],    references: [id])
  product   Product @relation(fields: [productId], references: [id])
  @@unique([userId, productId])
}

model Notification {
  id        Int      @id @default(autoincrement())
  message   String
  read      Boolean  @default(false)
  createdAt DateTime @default(now())
  userId    Int
  user      User     @relation(fields: [userId], references: [id])
}

model UserSettings {
  id     Int    @id @default(autoincrement())
  userId Int    @unique
  theme  String @default("dark")
  user   User   @relation(fields: [userId], references: [id])
}

🎯 Итоговые задания для защиты

  1. Базовый уровень (обязательно)
    Соберите приложение: запустите оба сервера, зарегистрируйтесь, добавьте 2 товара в корзину, примените промокод SAVE20, оформите заказ. Покажите преподавателю работающее приложение.
  2. Страница истории заказов
    Добавьте модель Order в Prisma (userId, total, items, createdAt). При нажатии «Оформить заказ» — создавать Order, очищать корзину, создавать уведомление. Добавить страницу /orders.
  3. Поиск и фильтрация
    Добавьте поле поиска над каталогом. Используйте GET /api/products?q=кроссовки — на бэкенде where: { name: { contains: q, mode: 'insensitive' } }.
  4. Пагинация
    Измените GET /api/products — принимает ?page=1&limit=6. На бэкенде: skip: (page-1)*limit, take: limit. Добавьте кнопки «Назад» / «Вперёд» в CatalogPage.
  5. ⭐ Избранное
    Добавьте модель Favorite (userId, productId, @@unique). Кнопка-сердечко на карточке товара. Страница /favorites со списком избранного.
  6. ⭐⭐ Docker Compose
    Напишите docker-compose.yml с тремя сервисами: postgres, backend, frontend. Приложение должно запускаться одной командой docker compose up.
