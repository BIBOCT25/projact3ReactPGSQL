Часть 1 — Фича #18

🔐 Индикатор силы пароля

Зачем: слабые пароли — главная причина взломов аккаунтов. Индикатор силы наглядно подсказывает пользователю ещё в форме, не дожидаясь ответа сервера. Backend хранит только bcrypt-хеш — никогда не сам пароль.
1

Логика оценки силы пароля

4 уровня: Слабый → Средний → Хороший → Сильный. Оценка полностью на фронтенде — нет смысла делать API.

Демо (интерактивное):

frontend/src/components/PasswordStrength.tsx
interface Props { password: string }

function getStrength(p: string): number {
  if (p.length < 6)  return 0  // Слабый
  if (p.length < 10) return 1  // Средний
  const hasUpper = /[A-Z]/.test(p)
  const hasDigit = /[0-9]/.test(p)
  const hasSpec  = /[^a-zA-Z0-9]/.test(p)
  if (hasUpper && hasDigit && hasSpec) return 3  // Сильный
  if (hasUpper && hasDigit)            return 2  // Хороший
  return 1
}

const LABELS = ['Слабый', 'Средний', 'Хороший', 'Сильный']
const COLORS = ['#f85149', '#f59e0b', '#3b82f6', '#10b981']

export function PasswordStrength({ password }: Props) {
  const strength = getStrength(password)
  const color    = COLORS[strength]

  if (!password) return null

  return (
    <div style={{ marginTop: 6 }}>
      <div style={{ display: 'flex', gap: 4 }}>
        {[0,1,2,3].map(i => (
          <div key={i} style={{
            height: 5, flex: 1, borderRadius: 3,
            background: i <= strength ? color : '#e5e7eb',
            transition: 'background 0.3s'
          }} />
        ))}
      </div>
      <span style={{ fontSize: 12, color }}>
        {LABELS[strength]}
      </span>
    </div>
  )
}

Хук useMiniCart + useClickOutside

Закрывать dropdown нужно при клике вне него. Для этого — отдельный хук:

frontend/src/hooks/useClickOutside.ts
import { useEffect, RefObject } from 'react'

export function useClickOutside(
  ref: RefObject<HTMLElement>,
  handler: () => void
) {
  useEffect(() => {
    const listener = (e: MouseEvent) => {
      if (!ref.current || ref.current.contains(e.target as Node)) return
      handler()
    }
    document.addEventListener('mousedown', listener)
    return () => document.removeEventListener('mousedown', listener)
  }, [ref, handler])
}

import { useRef, useState } from 'react'
import { useNavigate } from 'react-router-dom'
import { useClickOutside } from '../hooks/useClickOutside'
import { useCart } from '../hooks/useCart'

export function MiniCart() {
  const [open, setOpen]  = useState(false)
  const wrapRef          = useRef<HTMLDivElement>(null)
  const { cart, remove } = useCart()
  const navigate         = useNavigate()

  useClickOutside(wrapRef, () => setOpen(false))

  const count = cart?.items.reduce((s, i) => s + i.qty, 0) ?? 0

  return (
    <div ref={wrapRef} style={{ position: 'relative' }}>

      {/* Кнопка-триггер */}
      <button onClick={() => setOpen(o => !o)} style={{ position: 'relative' }}>
        🛒
        {count > 0 && (
          <span style={{
            position: 'absolute', top: -6, right: -6,
            background: '#ef4444', color: '#fff',
            borderRadius: '50%', width: 18, height: 18,
            fontSize: 11, fontWeight: 'bold',
            display: 'flex', alignItems: 'center', justifyContent: 'center'
          }}>
            {count > 99 ? '99+' : count}
          </span>
        )}
      </button>

      {/* Dropdown */}
      {open && (
        <div style={{
          position: 'absolute', right: 0, top: 'calc(100% + 8px)',

              </div>
              <button
                onClick={() => { setOpen(false); navigate('/checkout') }}

frontend/src/pages/CheckoutPage.tsx
import { useState, useEffect } from 'react'
import { useCart } from '../hooks/useCart'

interface DeliveryRate { id: number; name: string; price: number }

export function CheckoutPage() {
  const { cart }                  = useCart()
  const [rates, setRates]         = useState<DeliveryRate[]>([])
  const [selected, setSelected]   = useState<number | null>(null)

  useEffect(() => {
    fetch('/api/delivery-rates')
      .then(r => r.json())
      .then(data => {
        setRates(data)
        if (data.length) setSelected(data[0].id)   // первый по умолчанию
      })
  }, [])

  const delivery     = rates.find(r => r.id === selected)
  const deliveryCost = delivery?.price ?? 0
  const total        = (cart?.subtotal ?? 0) + deliveryCost

  const submit = async () => {
    await fetch('/api/orders', {
      method:  'POST',
      headers: {
        'Content-Type': 'application/json',
        Authorization: `Bearer ${localStorage.getItem('token')}`
      },
      body: JSON.stringify({ deliveryRateId: selected })
    })
  }

  return (
    <div>
      <h2>Способ доставки</h2>
      {rates.map(rate => (
        <label
          key={rate.id}
          style={{
            display: 'flex', gap: 12, padding: 14, cursor: 'pointer',
            border: `2px solid ${selected === rate.id ? '#bc8cff' : '#30363d'}`,
            borderRadius: 10, marginBottom: 8
          }}
        >
          <input
            type="radio"
            checked={selected === rate.id}
            onChange={() => setSelected(rate.id)}
          />
          <span>
            {rate.name} —{' '}
            <strong>
              {rate.price === 0 ? 'Бесплатно' : `${rate.price} ₽`}
            </strong>
          </span>
        </label>
      ))}
      <p>Итого с доставкой: <strong>{total} ₽</strong></p>
      <button onClick={submit}>Оформить</button>
    </div>
  )
}

router.get('/my', requireAuth, async (req, res) => {

// Детали одного заказа
router.get('/:id', requireAuth, async (req, res) => {
  const order = await prisma.order.findUnique({
    where:   { id: Number(req.params.id) },
    include: { items: { include: { product: true } } }
  })
  // Защита: заказ должен принадлежать текущему пользователю
  if (!order || order.userId !== req.userId)
    return res.status(404).json({ error: 'Не найден' })

  res.json(order)
})
IDOR-защита обязательна: всегда проверяй order.userId !== req.userId. Иначе любой залогиненный пользователь сможет посмотреть чужие заказы по id.

Часть 6 — UX: Toast-уведомления

🔔 Toast-уведомления (глобальный контекст)

Toast — всплывающее сообщение в углу экрана, исчезает через 3–4 секунды. Правило: один ToastContext на всё приложение, useToast() хук доступен в любом компоненте.

Демо:

✅ Товар добавлен в корзину
10

ToastContext + useToast хук

frontend/src/context/ToastContext.tsx
import { createContext, useContext, useState, useCallback } from 'react'

interface Toast { id: number; message: string; type: 'success' | 'error' | 'info' }

interface ToastCtx {
  toast: (message: string, type?: Toast['type']) => void
}

const ToastContext = createContext<ToastCtx>({} as ToastCtx)

export function ToastProvider({ children }: { children: React.ReactNode }) {
  const [toasts, setToasts] = useState<Toast[]>([])

  const toast = useCallback((message: string, type: Toast['type'] = 'info') => {
    const id = Date.now()
    setToasts(prev => [...prev, { id, message, type }])
    setTimeout(() => {
      setToasts(prev => prev.filter(t => t.id !== id))
    }, 3500)
  }, [])

  return (
    <ToastContext.Provider value={{ toast }}>
      {children}

      {/* Контейнер тостов */}
      <div style={{ position: 'fixed', bottom: 20, right: 20, zIndex: 9999, display: 'flex', flexDirection: 'column', gap: 8 }}>
        {toasts.map(t => (
          <div key={t.id} style={{
            background:   '#161b22',
            border:       `1px solid #30363d`,
            borderLeft:   `3px solid ${t.type === 'success' ? '#3fb950' : t.type === 'error' ? '#f85149' : '#58a6ff'}`,
            borderRadius: 10,
            padding:      '12px 20px',
            fontSize:     14,
            boxShadow:    '0 4px 20px rgba(0,0,0,.5)',
            animation:    'fadeInRight .3s ease'
          }}>
            {t.message}
          </div>
        ))}
      </div>
    </ToastContext.Provider>
  )
}

export const useToast = () => useContext(ToastContext)
